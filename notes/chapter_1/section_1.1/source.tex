\section{Discussion: the Irrationality of $\sqrt{2}$}
    \begin{theorem} 
        There is no rational number whose square is 2.
    \end{theorem}
    \begin{proof}
        \begin{equation}
            x = y + x
        \end{equation}
        \begin{equation}
            f(x) = x^2
        \end{equation}
        \begin{equation}
            f(x) = \sum_{i=1}^{n}{x_i} = x_1 + x_2 + \dots + x_n
        \end{equation}
    \end{proof}
